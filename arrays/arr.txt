    MERGE TWO SORTED ARRAYS
n1=int(input("enter the array1 size: "))
arr1=list(map(int,input().split()))
n2=int(input("enter the array2 size: "))
arr2=list(map(int,input().split()))
i=0
j=0
res=[]
while(i<len(arr1) and j<len(arr2)):
    if(arr1[i]<arr2[j]):
        res.append(arr1[i])
        i+=1
    else:
        res.append(arr2[j])
        j+=1
while(i<len(arr1)):
    res.append(arr1[i])
    i+=1
while(j<len(arr2)):
    res.append(arr2[j])
    j+=1
print(res)
